\newglossaryentry{drawdown}{name=maximum Drawdown, description={Le maximum Drawdown signifie la perte maximale historique qu'aurait subi un investisseur malchanceux s'il avait acheté au plus haut et revendu au plus bas durant un temps déterminé}}

\newglossaryentry{ruban}{name=ruban, description={Le ruban est une interface utilisateur graphique basée sur le principe des widgets, composée d'un bandeau qui regroupe toutes les fonctions du logiciel.
